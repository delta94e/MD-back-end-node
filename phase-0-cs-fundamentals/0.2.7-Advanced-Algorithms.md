## 1. Greedy Algorithms

### Ý tưởng

Tại mỗi bước, **chọn lựa chọn tốt nhất hiện tại** (locally optimal) mà không quay lại. Nếu bài có **greedy choice property**, local optimal → global optimal.

### 1.1 Activity Selection / Interval Scheduling

```javascript
// Chọn nhiều activities nhất không overlap
// GREEDY: sort theo end time, luôn chọn activity kết thúc sớm nhất
function maxActivities(intervals) {
  intervals.sort((a, b) => a[1] - b[1]); // sort by end time
  let count = 1,
    lastEnd = intervals[0][1];
  for (let i = 1; i < intervals.length; i++) {
    if (intervals[i][0] >= lastEnd) {
      // không overlap
      count++;
      lastEnd = intervals[i][1];
    }
  }
  return count;
}
```

### 1.2 Jump Game (LC #55)

```javascript
function canJump(nums) {
  let maxReach = 0;
  for (let i = 0; i < nums.length; i++) {
    if (i > maxReach) return false; // không thể đến đây
    maxReach = Math.max(maxReach, i + nums[i]);
  }
  return true;
}
// Greedy: track vị trí xa nhất có thể đạt được
```

### 1.3 Khi nào Greedy ĐÚNG?

```
✅ Greedy works:
  - Activity selection (sort by end time)
  - Huffman coding
  - Dijkstra (non-negative weights)
  - Coin change (specific denominations: 1, 5, 10, 25)
  - Jump Game

❌ Greedy FAILS:
  - 0/1 Knapsack → cần DP
  - Coin change (arbitrary denominations, ví dụ: [1, 3, 4] target=6)
    Greedy: 4+1+1=3 coins. Optimal: 3+3=2 coins!
  - Traveling Salesman → NP-hard
```

## 2. Backtracking

### Ý tưởng

**Thử tất cả khả năng**, nếu không thỏa → **quay lại** (backtrack) và thử lựa chọn khác. Bản chất là DFS trên "decision tree".

```
Template:
  function backtrack(state, choices):
    if isGoal(state):
      result.push(copy(state))
      return
    for choice in choices:
      if isValid(choice):
        make(choice)        // choose
        backtrack(...)      // explore
        undo(choice)        // un-choose (backtrack!)
```

### 2.1 Subsets (LC #78)

```javascript
function subsets(nums) {
  const result = [];
  function backtrack(start, current) {
    result.push([...current]); // mọi state đều là 1 subset
    for (let i = start; i < nums.length; i++) {
      current.push(nums[i]); // choose
      backtrack(i + 1, current); // explore
      current.pop(); // un-choose
    }
  }
  backtrack(0, []);
  return result;
}
// [1,2,3] → [],[1],[1,2],[1,2,3],[1,3],[2],[2,3],[3]
```

### 2.2 Permutations (LC #46)

```javascript
function permute(nums) {
  const result = [];
  function backtrack(current, remaining) {
    if (remaining.length === 0) {
      result.push([...current]);
      return;
    }
    for (let i = 0; i < remaining.length; i++) {
      current.push(remaining[i]);
      backtrack(current, [...remaining.slice(0, i), ...remaining.slice(i + 1)]);
      current.pop();
    }
  }
  backtrack([], nums);
  return result;
}
```

### 2.3 N-Queens (LC #51)

```javascript
function solveNQueens(n) {
  const result = [];
  const cols = new Set(),
    diag1 = new Set(),
    diag2 = new Set();
  const board = Array.from({ length: n }, () => ".".repeat(n));

  function backtrack(row) {
    if (row === n) {
      result.push([...board]);
      return;
    }
    for (let col = 0; col < n; col++) {
      if (cols.has(col) || diag1.has(row - col) || diag2.has(row + col))
        continue;
      // Place queen
      cols.add(col);
      diag1.add(row - col);
      diag2.add(row + col);
      board[row] =
        board[row].substring(0, col) + "Q" + board[row].substring(col + 1);
      backtrack(row + 1);
      // Remove queen (backtrack)
      cols.delete(col);
      diag1.delete(row - col);
      diag2.delete(row + col);
      board[row] =
        board[row].substring(0, col) + "." + board[row].substring(col + 1);
    }
  }
  backtrack(0);
  return result;
}
// Dùng 3 Sets để check conflicts trong O(1)
// Pruning: skip columns/diagonals đã có queen → giảm search space đáng kể
```

## 3. String Algorithms

### 3.1 KMP — Pattern Matching O(n+m)

```javascript
// Bước 1: Build failure function (prefix table)
function buildLPS(pattern) {
  const lps = new Array(pattern.length).fill(0);
  let len = 0,
    i = 1;
  while (i < pattern.length) {
    if (pattern[i] === pattern[len]) {
      len++;
      lps[i] = len;
      i++;
    } else {
      if (len !== 0)
        len = lps[len - 1]; // backtrack trong pattern
      else {
        lps[i] = 0;
        i++;
      }
    }
  }
  return lps;
}

// Bước 2: Search
function kmpSearch(text, pattern) {
  const lps = buildLPS(pattern);
  const result = [];
  let i = 0,
    j = 0; // i = text index, j = pattern index
  while (i < text.length) {
    if (text[i] === pattern[j]) {
      i++;
      j++;
    }
    if (j === pattern.length) {
      result.push(i - j); // found at index i-j
      j = lps[j - 1]; // continue searching
    } else if (i < text.length && text[i] !== pattern[j]) {
      if (j !== 0)
        j = lps[j - 1]; // skip matched prefix
      else i++;
    }
  }
  return result;
}
// O(n + m) — so với brute force O(n × m)
```

### 3.2 Rabin-Karp — Rolling Hash

```javascript
function rabinKarp(text, pattern) {
  const BASE = 256,
    MOD = 101;
  const n = text.length,
    m = pattern.length;
  let patHash = 0,
    txtHash = 0,
    h = 1;

  for (let i = 0; i < m - 1; i++) h = (h * BASE) % MOD;
  for (let i = 0; i < m; i++) {
    patHash = (BASE * patHash + pattern.charCodeAt(i)) % MOD;
    txtHash = (BASE * txtHash + text.charCodeAt(i)) % MOD;
  }

  const result = [];
  for (let i = 0; i <= n - m; i++) {
    if (patHash === txtHash) {
      // hash match → verify
      if (text.substring(i, i + m) === pattern) result.push(i);
    }
    if (i < n - m) {
      // slide window
      txtHash =
        (BASE * (txtHash - text.charCodeAt(i) * h) + text.charCodeAt(i + m)) %
        MOD;
      if (txtHash < 0) txtHash += MOD;
    }
  }
  return result;
}
// Average O(n + m), Worst O(n × m) khi nhiều hash collisions
// Ưu điểm: dễ extend cho multiple pattern search
```

## 4. Bit Manipulation

### 4.1 Operators cơ bản

```javascript
// AND (&)  — cả 2 bit = 1 → 1
// OR  (|)  — ít nhất 1 bit = 1 → 1
// XOR (^)  — khác nhau → 1, giống → 0
// NOT (~)  — đảo tất cả bits
// LEFT SHIFT  (<<)  — nhân 2^n
// RIGHT SHIFT (>>)  — chia 2^n

// XOR properties (CỰC QUAN TRỌNG):
// a ^ 0 = a
// a ^ a = 0
// a ^ b ^ a = b  (self-inverse!)
```

### 4.2 Common Tricks

```javascript
// 1. Check power of 2
function isPowerOfTwo(n) {
  return n > 0 && (n & (n - 1)) === 0;
}
// n = 8:  1000 & 0111 = 0000 ✓
// n = 6:  0110 & 0101 = 0100 ✗

// 2. Count set bits (Kernighan's)
function countBits(n) {
  let count = 0;
  while (n) {
    n &= n - 1;
    count++;
  } // clear lowest set bit
  return count;
}

// 3. Single Number — tìm số xuất hiện 1 lần (LC #136)
function singleNumber(nums) {
  return nums.reduce((xor, n) => xor ^ n, 0);
}
// [2, 1, 4, 1, 2] → 2^1^4^1^2 = (2^2)^(1^1)^4 = 0^0^4 = 4

// 4. Get/Set/Clear specific bit
function getBit(n, i) {
  return (n >> i) & 1;
}
function setBit(n, i) {
  return n | (1 << i);
}
function clearBit(n, i) {
  return n & ~(1 << i);
}

// 5. Swap without temp
function swap(a, b) {
  a ^= b;
  b ^= a;
  a ^= b;
  return [a, b];
}
```
