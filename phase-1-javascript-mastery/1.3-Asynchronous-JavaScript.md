## 1. Callbacks

### 1.1 Error-First Convention (Node.js pattern)

```javascript
const fs = require("fs");

// Error-first: callback(error, result)
fs.readFile("/path/to/file", "utf8", (err, data) => {
  if (err) {
    console.error("Error:", err.message);
    return;
  }
  console.log(data);
});
```

### 1.2 Callback Hell

```javascript
// ❌ Pyramid of doom
getUser(userId, (err, user) => {
  if (err) return handleError(err);
  getOrders(user.id, (err, orders) => {
    if (err) return handleError(err);
    getOrderDetails(orders[0].id, (err, details) => {
      if (err) return handleError(err);
      getShipping(details.shippingId, (err, shipping) => {
        if (err) return handleError(err);
        console.log(shipping);
      });
    });
  });
});

// Vấn đề:
// 1. Khó đọc (nesting sâu)
// 2. Error handling lặp lại
// 3. "Inversion of Control" — bạn TIN callback sẽ được gọi đúng
```

## 2. Promises — Deep Dive

### 2.1 Promise States

```
                    ┌─── fulfill(value) ───▶ FULFILLED ──▶ .then(onFulfilled)
                    │
PENDING ────────────┤
                    │
                    └─── reject(reason) ───▶ REJECTED ──▶ .catch(onRejected)

Rules:
  1. Promise starts as PENDING
  2. Once settled (fulfilled/rejected) → IMMUTABLE (không đổi lại)
  3. .then() LUÔN return new Promise (chaining)
  4. .then(onFulfilled, onRejected) — 2 params
  5. Unhandled rejection → UnhandledPromiseRejection (Node.js crash!)
```

### 2.2 Chaining & Error Propagation

```javascript
fetch("/api/users")
  .then((res) => res.json()) // return Promise<json>
  .then((users) => users[0]) // return value → auto-wrap in Promise
  .then((user) => fetch(`/api/orders/${user.id}`)) // return Promise<Response>
  .then((res) => res.json())
  .then((orders) => console.log(orders))
  .catch((err) => console.error(err)); // catches ANY error above!

// Error propagation:
// .then() → if handler throws → returned Promise REJECTED
// .then() → if handler returns rejected Promise → propagated
// Error "skip" through .then() cho đến khi gặp .catch()
```

### 2.3 Promise Static Methods

```javascript
// Promise.all — TẤT CẢ phải succeed. Fail-fast trên reject đầu tiên.
const results = await Promise.all([
  fetch("/api/users"),
  fetch("/api/orders"),
  fetch("/api/products"),
]);
// 1 fail → TẤT CẢ fail!

// Promise.allSettled — CHỜ TẤT CẢ, kể cả rejected
const outcomes = await Promise.allSettled([p1, p2, p3]);
// [{status:'fulfilled',value:...}, {status:'rejected',reason:...}, ...]
// Dùng khi: muốn biết kết quả từng promise

// Promise.race — NHANH NHẤT (fulfill/reject đầu tiên) wins
const fastest = await Promise.race([
  fetch("/api/data"),
  new Promise((_, reject) => setTimeout(() => reject("Timeout"), 5000)),
]);
// Dùng cho: timeout pattern

// Promise.any — NHANH NHẤT THÀNH CÔNG wins
const first = await Promise.any([server1, server2, server3]);
// Tất cả fail → AggregateError
// Dùng cho: redundant requests, fastest mirror
```

## 3. async/await

### 3.1 Bản chất

```javascript
// async/await = syntactic sugar over Promises + Generators

async function fetchUser(id) {
  const res = await fetch(`/api/users/${id}`);
  const user = await res.json();
  return user;
}

// TƯƠNG ĐƯƠNG:
function fetchUser(id) {
  return fetch(`/api/users/${id}`).then((res) => res.json());
}

// async function LUÔN return Promise
// await PAUSE execution cho đến khi Promise settled
```

### 3.2 Error Handling Patterns

```javascript
// Pattern 1: try/catch (phổ biến nhất)
async function getUser(id) {
  try {
    const res = await fetch(`/api/users/${id}`);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    console.error("Failed:", err.message);
    throw err; // re-throw hoặc return default
  }
}

// Pattern 2: Go-style [error, result] wrapper
function to(promise) {
  return promise.then((data) => [null, data]).catch((err) => [err, null]);
}

const [err, user] = await to(getUser(1));
if (err) {
  /* handle */
}

// Pattern 3: .catch() trên await expression
const user = await getUser(1).catch((err) => null);
```

### 3.3 Sequential vs Parallel — QUAN TRỌNG!

```javascript
// ❌ SEQUENTIAL (chậm!) — await lần lượt
async function slow() {
  const users = await fetch("/api/users"); // 200ms
  const orders = await fetch("/api/orders"); // 300ms
  const products = await fetch("/api/products"); // 150ms
  // Total: 650ms (sequential!)
}

// ✅ PARALLEL (nhanh!) — fire all, await together
async function fast() {
  const [users, orders, products] = await Promise.all([
    fetch("/api/users"), // 200ms ┐
    fetch("/api/orders"), // 300ms ├── parallel!
    fetch("/api/products"), // 150ms ┘
  ]);
  // Total: 300ms (max of all = bottleneck)
}
```

## 4. Concurrency Patterns

### 4.1 Promise Pool (Limit Concurrent)

```javascript
// Khi có 1000 URLs mà chỉ muốn fetch 5 cùng lúc
async function promisePool(tasks, concurrency) {
  const results = [];
  const executing = new Set();

  for (const task of tasks) {
    const p = task().then((result) => {
      executing.delete(p);
      return result;
    });
    results.push(p);
    executing.add(p);

    if (executing.size >= concurrency) {
      await Promise.race(executing); // wait for 1 slot to free up
    }
  }

  return Promise.all(results);
}

// Usage:
const urls = Array.from({ length: 100 }, (_, i) => `https://api.com/${i}`);
const tasks = urls.map((url) => () => fetch(url).then((r) => r.json()));
const results = await promisePool(tasks, 5); // max 5 concurrent
```

### 4.2 AbortController — Cancel Requests

```javascript
const controller = new AbortController();
const { signal } = controller;

// Cancel after 5 seconds
setTimeout(() => controller.abort(), 5000);

try {
  const res = await fetch("/api/slow-endpoint", { signal });
  const data = await res.json();
} catch (err) {
  if (err.name === "AbortError") {
    console.log("Request was cancelled");
  }
}
```

### 4.3 Debounce & Throttle

```javascript
// DEBOUNCE: chờ user NGỪNG gõ rồi mới gọi API
function debounce(fn, delay) {
  let timer;
  return function (...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

// THROTTLE: gọi TỐI ĐA 1 lần mỗi delay ms
function throttle(fn, delay) {
  let lastTime = 0;
  return function (...args) {
    const now = Date.now();
    if (now - lastTime >= delay) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}
```

## 5. Implement Promise From Scratch

```javascript
class MyPromise {
  constructor(executor) {
    this.state = "pending";
    this.value = undefined;
    this.callbacks = [];

    const resolve = (value) => {
      if (this.state !== "pending") return;
      this.state = "fulfilled";
      this.value = value;
      this.callbacks.forEach((cb) => cb.onFulfilled(value));
    };

    const reject = (reason) => {
      if (this.state !== "pending") return;
      this.state = "rejected";
      this.value = reason;
      this.callbacks.forEach((cb) => cb.onRejected(reason));
    };

    try {
      executor(resolve, reject);
    } catch (err) {
      reject(err);
    }
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      const handle = (handler, fallback) => (value) => {
        try {
          if (typeof handler === "function") {
            const result = handler(value);
            if (result instanceof MyPromise) {
              result.then(resolve, reject); // unwrap nested Promise
            } else {
              resolve(result);
            }
          } else {
            fallback(value);
          }
        } catch (err) {
          reject(err);
        }
      };

      const callback = {
        onFulfilled: handle(onFulfilled, resolve),
        onRejected: handle(onRejected, reject),
      };

      if (this.state === "pending") {
        this.callbacks.push(callback);
      } else if (this.state === "fulfilled") {
        queueMicrotask(() => callback.onFulfilled(this.value));
      } else {
        queueMicrotask(() => callback.onRejected(this.value));
      }
    });
  }

  catch(onRejected) {
    return this.then(null, onRejected);
  }

  finally(onFinally) {
    return this.then(
      (value) => MyPromise.resolve(onFinally()).then(() => value),
      (reason) =>
        MyPromise.resolve(onFinally()).then(() => {
          throw reason;
        }),
    );
  }

  static resolve(value) {
    return new MyPromise((resolve) => resolve(value));
  }

  static reject(reason) {
    return new MyPromise((_, reject) => reject(reason));
  }

  static all(promises) {
    return new MyPromise((resolve, reject) => {
      const results = [];
      let count = 0;
      promises.forEach((p, i) => {
        MyPromise.resolve(p).then((value) => {
          results[i] = value;
          if (++count === promises.length) resolve(results);
        }, reject);
      });
    });
  }

  static race(promises) {
    return new MyPromise((resolve, reject) => {
      promises.forEach((p) => MyPromise.resolve(p).then(resolve, reject));
    });
  }
}
```

## 6. Câu Hỏi Phỏng Vấn

**Q: Promise.all vs Promise.allSettled — khi nào dùng?**

> A: **Promise.all**: khi TẤT CẢ phải thành công (ví dụ: load critical data). 1 fail → tất cả fail. **Promise.allSettled**: khi muốn xử lý kết quả từng promise riêng (ví dụ: batch operations, some may fail). Trả về array `{status, value/reason}`.

**Q: Implement async/await bằng generators?**

> A: `async function` ~ generator + auto-runner. `await` ~ `yield`. Runner tự động `.next()` khi Promise resolves. `function run(gen) { const g = gen(); function step(val) { const {value, done} = g.next(val); if (done) return value; return Promise.resolve(value).then(step); } return step(); }`
