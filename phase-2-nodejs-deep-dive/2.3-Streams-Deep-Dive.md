## 1. 4 Loại Streams

```
┌──────────────────────────────────────────────────────┐
│                   NODE.JS STREAMS                     │
│                                                      │
│  ┌─────────────┐          ┌─────────────┐           │
│  │  READABLE   │─────────▶│  WRITABLE   │           │
│  │ (source)    │          │ (destination)│           │
│  │ fs.createRS │          │ fs.createWS  │           │
│  │ http.req    │          │ http.res     │           │
│  └─────────────┘          └─────────────┘           │
│                                                      │
│  ┌─────────────┐          ┌─────────────┐           │
│  │   DUPLEX    │          │  TRANSFORM  │           │
│  │ (read+write)│          │ (modify data│           │
│  │ net.Socket  │          │  in transit)│           │
│  │ WebSocket   │          │ zlib.gzip   │           │
│  └─────────────┘          │ crypto      │           │
│                           └─────────────┘           │
└──────────────────────────────────────────────────────┘
```

```javascript
const fs = require("fs");

// Readable → Writable (copy file)
const readable = fs.createReadStream("input.txt");
const writable = fs.createWriteStream("output.txt");
readable.pipe(writable);
// Data flows: input.txt → chunks → output.txt
// Memory: chỉ 1 chunk tại 1 thời điểm (64KB default)!
```

## 2. Backpressure — Cơ Chế Quan Trọng Nhất

```
Vấn đề: Source NHANH hơn Destination

  READABLE (SSD: 500MB/s)  ──▶  WRITABLE (Network: 10MB/s)

  Không có backpressure:
    Reader liên tục đọc 500MB/s
    → Buffer trong memory TRÀN → Out of Memory!

  Có backpressure:
    Writer: "Tôi đầy rồi!" (buffer > highWaterMark)
    → .write() return FALSE
    → Reader PAUSE
    → Writer xử lý xong → emit 'drain'
    → Reader RESUME
```

```javascript
// Manual backpressure handling:
const readable = fs.createReadStream("huge-file.csv");
const writable = fs.createWriteStream("output.csv");

readable.on("data", (chunk) => {
  const canContinue = writable.write(chunk);
  if (!canContinue) {
    // Buffer đầy! Pause reader
    readable.pause();
    writable.once("drain", () => {
      readable.resume(); // Buffer trống, tiếp tục
    });
  }
});

readable.on("end", () => writable.end());

// ✅ pipe() / pipeline() xử lý backpressure TỰ ĐỘNG!
// → LUÔN dùng pipeline() thay vì manual handling
```

## 3. Pipeline & Error Handling

```javascript
const { pipeline } = require("stream");
const { promisify } = require("util");
const pipelineAsync = promisify(pipeline);

// ❌ pipe() — KHÔNG handle errors!
readable.pipe(transform).pipe(writable);
// Nếu transform lỗi → memory leak! pipe không cleanup

// ✅ pipeline() — handle errors + cleanup
pipeline(
  fs.createReadStream("input.csv"),
  zlib.createGzip(),
  fs.createWriteStream("output.csv.gz"),
  (err) => {
    if (err) console.error("Pipeline failed:", err);
    else console.log("Pipeline complete");
  },
);

// ✅ Async/await version
try {
  await pipelineAsync(
    fs.createReadStream("input.csv"),
    transformStream,
    fs.createWriteStream("output.csv"),
  );
} catch (err) {
  console.error("Pipeline failed:", err);
}
```

## 4. Custom Streams

```javascript
const { Readable, Writable, Transform } = require("stream");

// ===== Custom Readable =====
class DatabaseReader extends Readable {
  constructor(options) {
    super({ ...options, objectMode: true });
    this.page = 0;
    this.pageSize = 100;
  }
  async _read() {
    const rows = await db.query(
      `SELECT * LIMIT ${this.pageSize} OFFSET ${this.page * this.pageSize}`,
    );
    if (rows.length === 0) {
      this.push(null); // Signal end of stream
    } else {
      rows.forEach((row) => this.push(row));
      this.page++;
    }
  }
}

// ===== Custom Transform (CSV Parser) =====
class CSVParser extends Transform {
  constructor() {
    super({ objectMode: true }); // output objects
    this.remainder = "";
  }
  _transform(chunk, encoding, callback) {
    const data = this.remainder + chunk.toString();
    const lines = data.split("\n");
    this.remainder = lines.pop(); // incomplete line → save
    for (const line of lines) {
      const fields = line.split(",");
      this.push({ name: fields[0], age: parseInt(fields[1]) });
    }
    callback();
  }
  _flush(callback) {
    if (this.remainder) {
      const fields = this.remainder.split(",");
      this.push({ name: fields[0], age: parseInt(fields[1]) });
    }
    callback();
  }
}

// ===== Custom Writable =====
class DatabaseWriter extends Writable {
  constructor() {
    super({ objectMode: true, highWaterMark: 100 }); // buffer 100 objects
    this.batch = [];
  }
  async _write(record, encoding, callback) {
    this.batch.push(record);
    if (this.batch.length >= 50) {
      await db.insertMany(this.batch);
      this.batch = [];
    }
    callback();
  }
  async _final(callback) {
    if (this.batch.length > 0) await db.insertMany(this.batch);
    callback();
  }
}

// ETL Pipeline: DB → CSV Parse → DB
await pipelineAsync(
  new DatabaseReader(),
  new CSVParser(),
  new DatabaseWriter(),
);
```

## 5. Real-World Patterns

```javascript
// 1. File upload (multi-GB) without running out of memory
app.post("/upload", (req, res) => {
  const writeStream = fs.createWriteStream(`/uploads/${req.headers.filename}`);
  req.pipe(writeStream); // req IS a Readable stream!
  writeStream.on("finish", () => res.json({ ok: true }));
});

// 2. HTTP response compression
const server = http.createServer((req, res) => {
  res.writeHead(200, { "Content-Encoding": "gzip" });
  pipeline(
    fs.createReadStream("big-data.json"),
    zlib.createGzip(),
    res, // res IS a Writable stream!
    (err) => {
      if (err) console.error(err);
    },
  );
});

// 3. Real-time log processing
pipeline(
  fs.createReadStream("/var/log/app.log"),
  new Transform({
    transform(chunk, enc, cb) {
      const lines = chunk.toString().split("\n");
      for (const line of lines) {
        if (line.includes("ERROR")) this.push(line + "\n");
      }
      cb();
    },
  }),
  fs.createWriteStream("errors.log"),
  (err) => {},
);
```

## 6. Câu Hỏi Phỏng Vấn

**Q: Backpressure là gì? Tại sao quan trọng?**

> A: Khi source nhanh hơn destination, data tích lũy trong buffer → OOM. Backpressure = cơ chế feedback: writable.write() return false → pause source → drain event → resume. pipeline() handle tự động. Thiếu backpressure = memory leak trong production (upload files, ETL, proxy).

**Q: pipe() vs pipeline() — tại sao dùng pipeline()?**

> A: `pipe()` KHÔNG handle errors — nếu middle stream lỗi, streams khác KHÔNG cleanup (memory leak, file descriptor leak). `pipeline()` auto-cleanup tất cả streams khi lỗi + callback/promise khi hoàn thành.
