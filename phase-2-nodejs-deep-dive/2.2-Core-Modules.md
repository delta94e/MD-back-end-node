## 1. fs — File System

```javascript
const fs = require("fs");
const fsp = require("fs/promises"); // Promise API

// ===== 3 cách dùng: Sync / Callback / Promise =====

// Sync — BLOCK event loop! Chỉ dùng lúc startup.
const data = fs.readFileSync("config.json", "utf8");

// Callback — non-blocking, dùng thread pool
fs.readFile("data.txt", "utf8", (err, data) => {
  if (err) throw err;
  console.log(data);
});

// Promise — modern, dùng với async/await
const content = await fsp.readFile("data.txt", "utf8");

// ===== Streaming — cho files lớn =====
const readStream = fs.createReadStream("huge-file.csv", {
  encoding: "utf8",
  highWaterMark: 64 * 1024, // 64KB chunks
});
readStream.on("data", (chunk) => processChunk(chunk));
readStream.on("end", () => console.log("Done"));

// ===== Watch file changes =====
fs.watch("./src", { recursive: true }, (eventType, filename) => {
  console.log(`${eventType}: ${filename}`);
});
// ⚠️ fs.watch không ổn định cross-platform → dùng chokidar trong production
```

## 2. http / https

```javascript
const http = require("http");

// Raw HTTP server (không Express!)
const server = http.createServer((req, res) => {
  // req = Readable Stream (IncomingMessage)
  // res = Writable Stream (ServerResponse)

  if (req.method === "POST" && req.url === "/api/data") {
    let body = "";
    req.on("data", (chunk) => (body += chunk));
    req.on("end", () => {
      const data = JSON.parse(body);
      res.writeHead(201, { "Content-Type": "application/json" });
      res.end(JSON.stringify({ id: 1, ...data }));
    });
  }
});

server.listen(3000);

// ===== http.Agent — Connection Pooling =====
const agent = new http.Agent({
  keepAlive: true, // reuse TCP connections
  maxSockets: 10, // max 10 concurrent connections per host
  maxFreeSockets: 5, // keep 5 idle connections
  timeout: 30000, // connection timeout
});
// Default: keepAlive=false → new TCP connection per request!
// Production: LUÔN dùng keepAlive: true
```

## 3. net — TCP

```javascript
const net = require("net");

// TCP Server
const server = net.createServer((socket) => {
  // socket = Duplex Stream (read + write)
  console.log("Client connected:", socket.remoteAddress);

  socket.on("data", (data) => {
    socket.write(`Echo: ${data}`); // gửi lại client
  });

  socket.on("end", () => console.log("Client disconnected"));
  socket.on("error", (err) => console.error(err));
});

server.listen(4000);
// HTTP được xây dựng TRÊN net module!
// WebSocket upgrade bắt đầu từ net socket
```

## 4. buffer — Binary Data

```javascript
// Buffer = raw binary data trong Node.js (trước khi có TypedArray)

// Tạo buffer
const buf1 = Buffer.alloc(10); // 10 bytes, filled with 0
const buf2 = Buffer.allocUnsafe(10); // 10 bytes, uninitialized (NHANH nhưng có old data!)
const buf3 = Buffer.from("Hello", "utf8");
const buf4 = Buffer.from([0x48, 0x65, 0x6c, 0x6c, 0x6f]); // "Hello"

// Encoding conversion
const base64 = buf3.toString("base64"); // "SGVsbG8="
const hex = buf3.toString("hex"); // "48656c6c6f"

// Buffer vs String:
// Buffer: binary data, fixed size, mutable
// String: UTF-16, immutable, flexible size
// → Dùng Buffer cho: file I/O, network data, crypto, images

// ⚠️ Security: allocUnsafe có thể chứa sensitive data từ memory!
// → LUÔN dùng alloc() trừ khi bạn CHẮC CHẮN sẽ fill ngay
```

## 5. events — EventEmitter

```javascript
const EventEmitter = require("events");

// Custom EventEmitter
class OrderService extends EventEmitter {
  placeOrder(order) {
    // business logic...
    this.emit("orderPlaced", order); // fire event
    this.emit("notification", order.userId); // multiple events
  }
}

const service = new OrderService();
service.on("orderPlaced", (order) => saveToDatabase(order));
service.on("orderPlaced", (order) => sendEmail(order)); // multiple listeners!
service.once("orderPlaced", (order) => logFirstOrder(order)); // fire once only

// ⚠️ Memory leak: listeners tích lũy
service.setMaxListeners(20); // default: 10, warning nếu vượt
// Fix: removeListener / removeAllListeners khi done

// Error convention:
service.on("error", (err) => console.error(err));
// ⚠️ Nếu KHÔNG có error listener → process crash!
// EventEmitter 'error' event = special case
```

## 6. crypto

```javascript
const crypto = require("crypto");

// Hashing (one-way — KHÔNG decrypt được)
const hash = crypto.createHash("sha256").update("password").digest("hex");

// HMAC (hash + secret key)
const hmac = crypto
  .createHmac("sha256", "secret-key")
  .update("data")
  .digest("hex");

// Password hashing (dùng bcrypt/scrypt — KHÔNG dùng SHA trực tiếp!)
const salt = crypto.randomBytes(16).toString("hex");
crypto.scrypt("password", salt, 64, (err, derivedKey) => {
  const hash = derivedKey.toString("hex");
  // Store: salt + hash
});

// Random bytes (cho tokens, session IDs)
const token = crypto.randomBytes(32).toString("hex"); // 64 hex chars

// Symmetric encryption (AES-256-GCM)
const key = crypto.randomBytes(32);
const iv = crypto.randomBytes(16);
const cipher = crypto.createCipheriv("aes-256-gcm", key, iv);
let encrypted = cipher.update("secret data", "utf8", "hex");
encrypted += cipher.final("hex");
const authTag = cipher.getAuthTag();
```

## 7. path, os, zlib

```javascript
const path = require("path");
const os = require("os");
const zlib = require("zlib");

// path — cross-platform
path.join("/users", "alice", "docs"); // /users/alice/docs
path.resolve("./src", "../config.json"); // /absolute/path/config.json
path.basename("/users/alice/file.txt"); // "file.txt"
path.extname("file.txt"); // ".txt"
// ⚠️ ESM: __dirname không available → dùng import.meta.url
// import { dirname } from "path";
// import { fileURLToPath } from "url";
// const __dirname = dirname(fileURLToPath(import.meta.url));

// os — system info
os.cpus().length; // CPU cores (cho cluster fork count)
os.totalmem(); // total RAM
os.freemem(); // available RAM
os.platform(); // "linux", "darwin", "win32"

// zlib — compression/decompression
const { pipeline } = require("stream");
pipeline(
  fs.createReadStream("input.txt"),
  zlib.createGzip(), // compress
  fs.createWriteStream("input.txt.gz"),
  (err) => {
    if (err) console.error(err);
  },
);
```

## 8. Câu Hỏi Phỏng Vấn

**Q: fs sync vs async vs stream — khi nào dùng?**

> A: **Sync**: chỉ lúc startup (đọc config), KHÔNG BAO GIỜ trong request handler. **Async (callback/promise)**: files nhỏ-vừa, đọc toàn bộ vào memory. **Stream**: files lớn (>100MB), real-time processing, ETL pipelines — dùng memory cố định dù file bao lớn.

**Q: EventEmitter memory leak — cách detect và fix?**

> A: Default max 10 listeners/event. Vượt quá → `MaxListenersExceededWarning`. Nguyên nhân: subscribers không removeListener. Fix: (1) `once()` thay `on()`, (2) `removeListener()` khi done, (3) `removeAllListeners()` khi destroy. Detect: `emitter.listenerCount('event')`.
