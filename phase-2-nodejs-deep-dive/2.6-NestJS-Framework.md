```
│  │  │ Repository  │  │ Repository  │       │    │

Controller = route handling (thin, no business logic)
Provider/Service = business logic + data access
```

```typescript
// Module — nhóm related components
@Module({
  imports: [TypeOrmModule.forFeature([User])],
  controllers: [UserController],
  providers: [UserService],
  exports: [UserService], // available to other modules
})
export class UserModule {}

// Controller — route handling
@Controller("users")
export class UserController {
  constructor(private readonly userService: UserService) {} // DI!

  @Get(":id")
  findOne(@Param("id", ParseIntPipe) id: number) {
    return this.userService.findById(id);
  }

  @Post()
  @UseGuards(AuthGuard)
  create(@Body(ValidationPipe) dto: CreateUserDto) {
    return this.userService.create(dto);
  }
}

// Service — business logic
@Injectable()
export class UserService {
  constructor(
    @InjectRepository(User)
    private readonly userRepo: Repository<User>,
  ) {}

  async findById(id: number): Promise<User> {
    const user = await this.userRepo.findOneBy({ id });
    if (!user) throw new NotFoundException(`User #${id} not found`);
    return user;
  }
}
```

## 2. Dependency Injection

```
IoC Container (Inversion of Control):
  Thay vì: UserController tạo UserService
  NestJS:   Container TẠO UserService → INJECT vào Controller

Provider Scopes:
  DEFAULT     — Singleton (1 instance toàn app) ← production default
  REQUEST     — 1 instance per HTTP request (DB transaction scope)
  TRANSIENT   — New instance mỗi lần inject
```

```typescript
// Custom Providers
@Module({
  providers: [
    // useClass — standard
    { provide: UserService, useClass: UserService },

    // useValue — constant/mock
    { provide: "API_KEY", useValue: "abc123" },

    // useFactory — dynamic creation
    {
      provide: "DB_CONNECTION",
      useFactory: async (config: ConfigService) => {
        return createConnection(config.get("DATABASE_URL"));
      },
      inject: [ConfigService],
    },

    // useExisting — alias
    { provide: "AliasService", useExisting: UserService },
  ],
})
export class AppModule {}

// Inject custom provider:
constructor(@Inject("API_KEY") private apiKey: string) {}
```

## 3. Request Lifecycle

```
Request đến NestJS — thứ tự xử lý:

  Client Request
       │
       ▼
  ① MIDDLEWARE          (logging, CORS, body parsing)
       │
       ▼
  ② GUARDS              (authentication, authorization)
       │                 canActivate() → true/false
       ▼
  ③ INTERCEPTORS (before)(logging, caching, transform request)
       │
       ▼
  ④ PIPES               (validation, transformation)
       │                 transform/validate @Body, @Param
       ▼
  ⑤ ROUTE HANDLER       (controller method)
       │
       ▼
  ⑥ INTERCEPTORS (after) (transform response, timing)
       │
       ▼
  ⑦ EXCEPTION FILTERS   (catch errors, format response)
       │
       ▼
  Client Response

Nếu exception ở bất kỳ bước nào → nhảy đến ⑦ Exception Filters
```

## 4. Guards, Interceptors, Pipes, Filters

```typescript
// GUARD — authentication/authorization
@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}
  canActivate(context: ExecutionContext): boolean {
    const roles = this.reflector.get<string[]>("roles", context.getHandler());
    if (!roles) return true;
    const request = context.switchToHttp().getRequest();
    return roles.includes(request.user?.role);
  }
}
// Usage: @SetMetadata("roles", ["admin"]) + @UseGuards(RolesGuard)

// INTERCEPTOR — AOP (cross-cutting concerns)
@Injectable()
export class LoggingInterceptor implements NestInterceptor {
  intercept(context: ExecutionContext, next: CallHandler) {
    const now = Date.now();
    return next.handle().pipe(tap(() => console.log(`${Date.now() - now}ms`)));
  }
}

// PIPE — validation + transformation
@Injectable()
export class ParseIntPipe implements PipeTransform<string, number> {
  transform(value: string): number {
    const val = parseInt(value, 10);
    if (isNaN(val)) throw new BadRequestException("Not a number");
    return val;
  }
}

// EXCEPTION FILTER
@Catch(HttpException)
export class HttpExceptionFilter implements ExceptionFilter {
  catch(exception: HttpException, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();
    const status = exception.getStatus();
    response.status(status).json({
      statusCode: status,
      message: exception.message,
      timestamp: new Date().toISOString(),
    });
  }
}
```

## 5. Modules — Dynamic & Feature

```typescript
// Dynamic Module — forRoot / forFeature pattern
@Module({})
export class DatabaseModule {
  static forRoot(options: DatabaseOptions): DynamicModule {
    return {
      module: DatabaseModule,
      global: true, // available EVERYWHERE
      providers: [
        { provide: "DB_OPTIONS", useValue: options },
        DatabaseService,
      ],
      exports: [DatabaseService],
    };
  }

  static forFeature(entities: Type[]): DynamicModule {
    return {
      module: DatabaseModule,
      providers: entities.map((entity) => ({
        provide: `${entity.name}_REPO`,
        useFactory: (db: DatabaseService) => db.getRepo(entity),
        inject: [DatabaseService],
      })),
      exports: entities.map((e) => `${e.name}_REPO`),
    };
  }
}

// Usage:
@Module({
  imports: [
    DatabaseModule.forRoot({ url: "postgres://..." }),  // AppModule — 1 lần
    DatabaseModule.forFeature([User, Order]),            // Feature module
  ],
})
```

## 6. Microservices & WebSocket

```typescript
// Microservice (TCP transport)
// main.ts
const app = await NestFactory.createMicroservice(AppModule, {
  transport: Transport.TCP,
  options: { port: 3001 },
});

// Handler
@Controller()
export class MathController {
  @MessagePattern({ cmd: "sum" })
  sum(data: number[]): number {
    return data.reduce((a, b) => a + b, 0);
  }

  @EventPattern("user_created")
  handleUserCreated(data: any) {
    console.log("Event:", data); // fire-and-forget
  }
}

// WebSocket Gateway
@WebSocketGateway()
export class ChatGateway {
  @SubscribeMessage("message")
  handleMessage(@MessageBody() data: string): string {
    return `Echo: ${data}`;
  }
}
```

## 8. Câu Hỏi Phỏng Vấn

**Q: NestJS request lifecycle — thứ tự xử lý?**

> A: Middleware → Guards → Interceptors (before) → Pipes → Route Handler → Interceptors (after) → Exception Filters. Guards quyết định access (canActivate), Pipes validate/transform input, Interceptors wrap execution (logging, caching), Filters catch exceptions.

**Q: DI Scope: DEFAULT vs REQUEST vs TRANSIENT?**

> A: **DEFAULT** (Singleton): 1 instance shared toàn app — hiệu quả nhất. **REQUEST**: new instance mỗi HTTP request — dùng cho request-scoped data (DB transaction, tenant context). **TRANSIENT**: new instance mỗi lần inject — hiếm dùng. ⚠️ REQUEST scope propagate: nếu A (request) inject B (default) → B cũng trở thành request-scoped!
